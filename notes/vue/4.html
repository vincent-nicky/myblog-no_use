    # 案例二：分类展示列表

    • 方法一：不使用computed
    <template>
  <div>
    <div>
      <button @click="showAll">全部</button>
      <button @click="showMale">男</button>
      <button @click="showFemale">女</button>
    </div>
    <ul>
      <li v-for="(u, index) in displayUsers" :key="index">
        {{ u.name }} - {{ u.gender }}
      </li>
    </ul>
  </div>
</template>
<script>
let id = 0;
const createUser = (name, gender) => {
  id += 1;
  return {id: id, name: name, gender: gender};
};
export default {
  name: "App",
  data() {
    return {
      users: [
        createUser("方方", "男"),
        createUser("圆圆", "女"),
        createUser("小新", "男"),
        createUser("小葵", "女"),
      ],
      displayUsers: [],
    };
  },
  created() {
    this.displayUsers = this.users;
  },
  methods: {
    showMale() {
      this.displayUsers = this.users.filter((u) => u.gender === "男");
    },
    showFemale() {
      this.displayUsers = this.users.filter((u) => u.gender === "女");
    },
    showAll() {
      this.displayUsers = this.users;
    },
  },
};
</script>

    • 方法二：使用computed
    <template>
  <div>
    <div>
      <button @click="setGender('')">全部</button>
      <button @click="setGender('male')">男</button>
      <button @click="setGender('female')">女</button>
    </div>
    <ul>
      <li v-for="(u, index) in displayUsers" :key="index">
        {{ u.nickname }} - {{ u.gender }}
      </li>
    </ul>
  </div>
</template>
<script>
export default {
  name: "App",
  data() {
    return {
      users: [
        {id: 1, nickname: "方方", gender: "男"},
        {id: 2, nickname: "圆圆", gender: "女"},
        {id: 3, nickname: "小新", gender: "男"},
        {id: 4, nickname: "小葵", gender: "女"},
      ],
      gender: "",
    };
  },
  computed: {
    displayUsers() {
      const hash = {
        mail: "男",
        femail: "女",
      };
      const {users, gender} = this;
      if (gender === "") {
        return users;
      } else if (typeof gender === "string") {
        return users.filter((u) => u.gender === hash[gender]);
      } else {
        throw new Error("gender 的值是意外的值");
      }
    },
  },
  methods: {
    setGender(string) {
      this.gender = string;
    },
  },
};
</script>

    # 2. watch的两个例子

    • 用途：当数据变化时，执行一个函数

    # 案例一：简单撤销的实现

    <template>
  <div>
    {{ n }}
    <hr />
    <button @click="add1">+1</button>
    <button @click="add2">+2</button>
    <button @click="minus1">-1</button>
    <button @click="minus2">-2</button>
    <hr />
    <button @click="undo">撤销</button>
    <hr />
    {{ history }}
  </div>
</template>
<script>
export default {
  name: "App",
  data() {
    return {
      n: 0,
      history: [],
      inUndoMode: false
    };
  },
  watch: {
    n: function (newValue, oldValue) {
      // console.log(this.inUndoMode);
      console.log(`在不在撤销模式: ${this.inUndoMode ? '在' : '不在'}`)
      if (!this.inUndoMode) {
        this.history.push({ from: oldValue, to: newValue });
      }
    }
  },
  methods: {
    add1() {
      this.n += 1;
    },
    add2() {
      this.n += 2;
    },
    minus1() {
      this.n -= 1;
    },
    minus2() {
      this.n -= 2;
    },
    undo() {
      const last = this.history.pop();
      console.log("ha" + this.inUndoMode);
      const old = last.from;
      this.inUndoMode = true;
      this.n = old; // watch n 的函数会异步调用
      // this.$nextTick(() => {
        this.inUndoMode = false;
      // });
    }
  }
};
</script>

    # 案例二：模拟computed

    <template>
  <div>
    {{ displayName }}
    <button @click="user.nickname = undefined">remove nickname</button>
  </div>
</template>

<script>
export default {
  name: "App",
  data() {
    return {
      user: {
        email: "fangfang@qq.com",
        nickname: "方方",
        phone: "13812312312"
      },
      displayName: "",
    }
  },
  watch: {
    "user.email": {
      handler: "changed",
      immediate: true // 第一次渲染时也触发 watch
    },
    "user.nickname": {
      handler: "changed",
      immediate: true // 第一次渲染是时触发 watch
    },
    "user.phone": {
      handler: "changed",
      immediate: true // 第一次渲染是时触发 watch
    }
  },
  methods: {
    changed() {
      console.log(arguments);
      // const user = this.user;
      // this.displayName = user.nickname || user.email || user.phone;
      
      //这里不能使用user
      const { user: { email, nickname, phone } } = this;
      this.displayName = nickname || email || phone;
    }
  }
};
</script>

    # 3. watch 的 deep 选项

    # 案例：通过watch可以探究 === 的规则

    <template>
  <div>
    <button @click="n += 1">n+1</button>
    <button @click="obj.a += 'hi'">obj.a + 'hi'</button>
    <button @click="obj = { a: 'a', b: 'b' }">obj = 新对象</button>
  </div>
</template>
<script>
export default {
  name: "App",
  data() {
    return {
      n: 0,
      obj: {
        a: "a",
        b: "b"
      }
    }
  },
  watch: {
    n() {
      console.log("n 变了");
    },
    obj: {
      handler() {
        console.log("obj 变了");
      },
      deep: true
    },
    "obj.a": function () {
      console.log("obj.a 变了");
    },
    "obj.b": function () {
      console.log("obj.b 变了");
    }
  },
};
</script>

    • deep的意思是：监听object的时候是否往深了看

    • 若有一个需求：只要obj里的值变了都算obj变了

    deep默认为false

    如果不加deep，若要判断obj是否变了，则需要把obj里的属性全部重新赋值，vue才知道obj变了，加了deep后，只要obj里的一个属性变了，则这个obj变了

    结论：简单类型看值，复杂类型（对象）看地址

    简单类型的值变了，watch可以发现它变了；但是obj里属性的值变了，watch只发现这个属性变了，但obj没有变。（默认情况下，简单类型的值变了则它就变了，obj的地址没有变，则obj没有变）

    # 5. computed VS watch (opens new window)

    1. 英文翻译
    2. 各自描述（引用代码例子）
      • computed： 1.调用时不需要加括号；2.缓存
      • watch： 1. immediate；deep 2.变化才执行

    computed 表示计算属性，watch 表示监听。

    computed是用来计算出一个值的，这个值在调用时不需要加括号，可以当作属性来用，它会根据依赖自动缓存，如果依赖的属性没有变化，computed就不会重新计算。

    watch 的用途是当数据变化时，执行一个函数。它有两个可选选项：immediate 和 deep，immediate 表示是否在第一次渲染时运行 watch，deep表示监听对象时是否要看对象的里面的变化。

    1.如果一个数据依赖于其他数据，那么把这个数据设计为computed的

    2.如果你需要在某个数据变化时做一些事情，使用watch来观察这个数据变化
